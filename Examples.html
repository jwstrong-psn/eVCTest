When you move the mouse over the text below, the tooltip text will Hover over me `name` as the input when creating a person fade in and take 1 second to go from completely invisible to visible.

var person = function(name) {
    Object.defineProperty(this, 'fullName', {
        get: function() {
            return this.firstName + ' ' + this.lastName;
        },
        set: function(name) {
            var words = name.split(' ');
            this.firstName = words[0] || '';
            this.lastName = words[1] || '';
        }
    });
    this.fullName = name;
}
